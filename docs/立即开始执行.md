## 🎨 前端实现

### 进游戏时检查日志

**文件：** `frontend/src/pages/Home.tsx` 或 `frontend/src/pages/Court.tsx`

```typescript
useEffect(() => {
  const checkUnviewedLogs = async () => {
    const response = await api.get(`/logs/unviewed/${courtId}/${userId}`);
    
    if (response.data.unviewedLogs.length > 0) {
      setShowLogModal(true);
      setUnviewedLogs(response.data.unviewedLogs);
    }
  };
  
  checkUnviewedLogs();
}, [courtId, userId]);
```

### 创建日志弹窗组件

**文件：** `frontend/src/components/LogModal.tsx`

```typescript
interface LogModalProps {
  logs: SarcasmLog[];
  onConfirm: () => void;
}

export const LogModal: React.FC<LogModalProps> = ({ logs, onConfirm }) => {
  const [currentIndex, setCurrentIndex] = useState(0);
  
  const handleConfirm = async () => {
    // 标记所有日志为已查看
    await api.post('/logs/mark-viewed', {
      courtId,
      userId,
      logIds: logs.map(log => log.id)
    });
    
    onConfirm();
  };
  
  return (
    <Modal>
      <div className="log-content">
        <div className="log-item">
          {logs[currentIndex].log_message}
        </div>
      </div>
      
      <div className="log-navigation">
        <button onClick={() => setCurrentIndex(currentIndex - 1)} disabled={currentIndex === 0}>
          上一条
        </button>
        <span>{currentIndex + 1} / {logs.length}</span>
        <button onClick={() => setCurrentIndex(currentIndex + 1)} disabled={currentIndex === logs.length - 1}>
          下一条
        </button>
      </div>
      
      <div className="log-actions">
        <button onClick={handleConfirm} className="confirm-btn">
          确认 ({logs.length} 条日志已全部查看)
        </button>
      </div>
    </Modal>
  );
};
```

## 🐛 常见问题

### Q: 迁移失败，提示权限不足？

A: 确保在 Supabase 控制台的 SQL Editor 中执行，不要用其他工具。

### Q: API 返回 404？

A: 确保后端已重启，路由已正确注册。

### Q: 日志没有创建？

A: 检查：
1. 数据库连接是否正常
2. 参数是否正确
3. 查看后端日志输出

### Q: 如何回滚迁移？

A: 在 SQL 编辑器执行：

```sql
DROP TRIGGER IF EXISTS trigger_update_unviewed_count ON emperor_sarcasm_logs;
DROP FUNCTION IF EXISTS update_unviewed_count();
DROP FUNCTION IF EXISTS mark_logs_as_viewed(UUID, UUID, UUID[]);
DROP FUNCTION IF EXISTS get_unviewed_logs(UUID, UUID, INT);
DROP FUNCTION IF EXISTS get_log_history(UUID, UUID, INT, INT);
DROP FUNCTION IF EXISTS create_sarcasm_log(UUID, UUID, VARCHAR, TEXT, JSONB);
DROP VIEW IF EXISTS unviewed_logs_view;
DROP INDEX IF EXISTS idx_emperor_logs_court_user;
DROP INDEX IF EXISTS idx_emperor_logs_created_at;
DROP INDEX IF EXISTS idx_emperor_logs_viewed;
DROP INDEX IF EXISTS idx_minister_progress_court_user;
DROP TABLE IF EXISTS minister_log_progress;
DROP TABLE IF EXISTS emperor_sarcasm_logs;
```
