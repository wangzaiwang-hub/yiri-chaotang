## 段子的随机性

```typescript
// 每次显示排行榜时，随机选择一个段子
const getRandomJoke = (level: string) => {
  const jokes = jokesMap[level];
  return jokes[Math.floor(Math.random() * jokes.length)];
};

// 每次任务失败时，随机选择一个段子
const getFailureJoke = (taskType: string) => {
  const jokes = failureJokesMap[taskType];
  return jokes[Math.floor(Math.random() * jokes.length)];
};

// 每次排名变化时，随机选择一个段子
const getRankChangeJoke = (oldRank: number, newRank: number) => {
  if (newRank < oldRank) {
    return "恭喜你，无能指数又创新高了！";
  } else {
    return "哟，你居然进步了？我很欣慰";
  }
};
```
